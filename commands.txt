CREATE TABLE products (
    id SERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    price NUMERIC(10,2) NOT NULL,
    image_url TEXT,
    sales_count INT DEFAULT 0,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 1. جدول الطلبات الرئيسي (Orders Table)
CREATE TABLE IF NOT EXISTS orders (
    id VARCHAR(50) PRIMARY KEY,        -- الـ order_id اللي بنولده كـ UUID نصوص
    user_id VARCHAR(50) NOT NULL,      -- الـ sub القادم من كوجنيتو للمستخدم
    total DECIMAL(10, 2) NOT NULL,     -- إجمالي المبلغ بصيغة رقمية دقيقة
    status VARCHAR(20) NOT NULL,       -- حالة الطلب (مثلاً COMPLETED)
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 2. جدول تفاصيل المنتجات جوه الطلب (Order Items Table)
CREATE TABLE IF NOT EXISTS order_items (
    id SERIAL PRIMARY KEY,             -- معرف تلقائي فريد لكل سطر
    order_id VARCHAR(50) NOT NULL,     -- ربط بجدول الطلبات الرئيسي
    product_id INT NOT NULL,           -- رقم المنتج (تم التحقق إنه Integer)
    quantity INT NOT NULL,             -- الكمية (تم التحقق إنها Integer موجبة)
    FOREIGN KEY (order_id) REFERENCES orders(id) ON DELETE CASCADE
);
